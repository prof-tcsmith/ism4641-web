import numpy as np
import time

# Performance comparison
size = 1000000

# Python list approach
python_list = list(range(size))
start_time = time.time()
python_result = [x * 2 for x in python_list]

numpy_array = np.arange(size)

Business Logic Applied to Arrays

# Pricing analysis with discount tiers
base_prices = np.array([999.99, 1299.99, 1599.99, 1999.99, 2499.99])

def apply_volume_discount(prices, quantity):
    """Apply volume discounts based on quantity purchased."""
    if quantity >= 100:
        return prices * 0.85  # 15% discount
    elif quantity >= 50:
        return prices * 0.90  # 10% discount
    elif quantity >= 10:
        return prices * 0.95  # 5% discount
    else:
        return prices  # No discount

def calculate_profit_margins(selling_prices, cost_ratio=0.60):
    """Calculate profit margins assuming costs are 60% of selling price."""
    costs = selling_prices * cost_ratio
    profits = selling_prices - costs
    margins = (profits / selling_prices) * 100
    return profits, margins

# Vectorized function application
print("PRICING STRATEGY ANALYSIS")
print("=" * 30)

print("Base Prices:", [f"${p:.2f}" for p in base_prices])

# Apply discounts for different quantities
for qty in [5, 25, 75, 150]:
    discounted_prices = apply_volume_discount(base_prices, qty)
    total_discount = base_prices.sum() - discounted_prices.sum()
    
    print(f"\nQuantity {qty} units:")
    print(f"  Total Value: ${discounted_prices.sum():,.2f}")
    print(f"  Total Discount: ${total_discount:,.2f}")
    
    # Calculate profitability
    profits, margins = calculate_profit_margins(discounted_prices)
    print(f"  Total Profit: ${profits.sum():,.2f}")
    print(f"  Average Margin: {margins.mean():.1f}%")

# Using numpy's vectorize for complex functions
def business_tier_discount(price):
    """Complex discount logic based on price tiers."""
    if price > 2000:
        return price * 0.80  # 20% off premium items
    elif price > 1500:
        return price * 0.85  # 15% off high-end items
    elif price > 1000:
        return price * 0.90  # 10% off mid-range items
    else:
        return price * 0.95  # 5% off entry-level items

# Vectorize the function for array operations
vectorized_discount = np.vectorize(business_tier_discount)
tiered_prices = vectorized_discount(base_prices)

print(f"\nTiered Pricing Strategy:")

print(f"Tiered:   {[f'${p:.2f}' for p in tiered_prices]}")
print(f"Total Savings: ${(base_prices - tiered_prices).sum():.2f}")

TechTrends Sales Analysis System

# Comprehensive business analytics using NumPy
import numpy as np

# Simulate TechTrends sales data
np.random.seed(42)  # For reproducible results

# Create realistic sales data with seasonal patterns
base_sales = 50000
months = 12
regions = 4

# Generate sales with growth trend and seasonality
month_multipliers = np.array([0.9, 0.85, 0.95, 1.0, 1.05, 1.1, 
                             1.15, 1.1, 1.05, 1.0, 0.95, 1.2])  # Holiday boost in Dec
regional_multipliers = np.array([1.2, 0.8, 1.5, 1.0])  # Different regional performance

# Create sales matrix
sales_matrix = np.zeros((regions, months))
for region in range(regions):
    for month in range(months):
        base = base_sales * regional_multipliers[region] * month_multipliers[month]
        # Add random variation Â±10%
        variation = np.random.normal(0, 0.1)
        sales_matrix[region, month] = base * (1 + variation)

print("TECHTRENDZ COMPREHENSIVE SALES ANALYSIS")
print("=" * 50)

regions_names = ['North', 'South', 'East', 'West'] 
months_names = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun',
               'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec']

# 1. Revenue Analysis
total_revenue = sales_matrix.sum()
regional_revenue = sales_matrix.sum(axis=1)
monthly_revenue = sales_matrix.sum(axis=0)

print("1. REVENUE ANALYSIS")
print("-" * 20)
print(f"Total Annual Revenue: ${total_revenue:,.2f}")
print("\nRegional Breakdown:")
for i, region in enumerate(regions_names):
    percentage = (regional_revenue[i] / total_revenue) * 100
    print(f"  {region}: ${regional_revenue[i]:,.2f} ({percentage:.1f}%)")

# 2. Growth Analysis
monthly_growth = np.diff(monthly_revenue) / monthly_revenue[:-1] * 100
print(f"\n2. GROWTH ANALYSIS")
print("-" * 18)
print("Month-over-Month Growth Rates:")
for i, growth in enumerate(monthly_growth):
    print(f"  {months_names[i]} to {months_names[i+1]}: {growth:+.1f}%")

best_growth_month = months_names[monthly_growth.argmax() + 1]
print(f"\nBest Growth Month: {best_growth_month} ({monthly_growth.max():.1f}%)")

# 3. Performance Rankings
regional_avg = sales_matrix.mean(axis=1)
performance_ranking = np.argsort(regional_avg)[::-1]  # Descending order

print(f"\n3. REGIONAL PERFORMANCE RANKING")
print("-" * 35)
for rank, region_idx in enumerate(performance_ranking):
    region = regions_names[region_idx]
    avg_sales = regional_avg[region_idx]
    print(f"  {rank+1}. {region}: ${avg_sales:,.2f} avg monthly")

# 4. Risk Analysis (Volatility)
regional_volatility = sales_matrix.std(axis=1) / sales_matrix.mean(axis=1)

print(f"\n4. VOLATILITY ANALYSIS")
print("-" * 22)
print("Regional Sales Volatility (Coefficient of Variation):")
for i, region in enumerate(regions_names):
    volatility = regional_volatility[i] * 100
    risk_level = "Low" if volatility < 10 else "Medium" if volatility < 20 else "High"
    print(f"  {region}: {volatility:.1f}% ({risk_level} Risk)")

# 5. Forecasting (Simple trend projection)
recent_trend = monthly_revenue[-3:].mean() - monthly_revenue[:3].mean()
monthly_trend = recent_trend / 9  # 9 months between periods

projected_next_month = monthly_revenue[-1] + monthly_trend
print(f"\n5. FORECASTING")
print("-" * 13)
print(f"Projected Next Month Revenue: ${projected_next_month:,.2f}")
print(f"Monthly Trend: ${monthly_trend:,.2f}")

# 6. Key Performance Indicators
print(f"\n6. KEY PERFORMANCE INDICATORS")
print("-" * 30)
print(f"Average Monthly Revenue: ${monthly_revenue.mean():,.2f}")
print(f"Revenue Standard Deviation: ${monthly_revenue.std():,.2f}")
print(f"Best Month: {months_names[monthly_revenue.argmax()]} (${monthly_revenue.max():,.2f})")
print(f"Worst Month: {months_names[monthly_revenue.argmin()]} (${monthly_revenue.min():,.2f})")
print(f"Revenue Range: ${monthly_revenue.max() - monthly_revenue.min():,.2f}")
