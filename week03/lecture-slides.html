Part 1: Tuples

Introduction to Tuples

Tuples are immutable, ordered collections

  • Ordered: Elements maintain their position
  • Immutable: Cannot be changed after creation
  • Hashable: Can be used as dictionary keys
  • Parentheses: Created using () or comma separation

Part 2: Sets

Introduction to Sets

Sets store unique, unordered collections

  • Unique Elements: Automatically removes duplicates
  • Unordered: No index-based access
  • Mutable: Can add/remove elements
  • Fast Lookups: O(1) average case for membership testing
# Creating sets
empty_set = set()  # Note: {} creates a dict, not a set
customer_ids = {101, 102, 103, 104, 105}
mixed_set = {1, "hello", 3.14, True}

# From lists (removes duplicates)
sales_reps = ["Alice", "Bob", "Charlie", "Alice", "Diana", "Bob"]
unique_reps = set(sales_reps)

print(f"Sales reps with duplicates: {sales_reps}")
print(f"Unique sales reps: {unique_reps}")
Sales reps with duplicates: ['Alice', 'Bob', 'Charlie', 'Alice', 'Diana', 'Bob']
Unique sales reps: {'Alice', 'Charlie', 'Diana', 'Bob'}

Part 3: Functions

Why Functions Matter

Functions are the building blocks of professional programming:

  • Code Reusability: Write once, use many times
  • Organization: Break complex problems into manageable pieces
  • Testing: Easier to test small, focused functions
  • Maintenance: Changes in one place affect all uses
  • Collaboration: Clear interfaces between team members

Business Benefits

  • Consistent calculations across reports
  • Reduced errors from copy-paste code
  • Faster development of new features
  • Easier to update business rules

# Function with parameters
def calculate_discount(price, discount_rate):
    """Calculate discounted price for a product.
    
    Args:
        price (float): Original price of the product
        discount_rate (float): Discount rate as decimal (0.1 = 10%)
    
    Returns:
        float: Discounted price
    """
    discounted_price = price * (1 - discount_rate)
    return discounted_price

# Using the function

print(f"You saved: ${original_price - final_price:.2f}")

Function Parameters

Default Parameters

def calculate_loan_payment(principal, annual_rate=0.05, years=30):
    """Calculate monthly loan payment using standard formula.
    
    Args:
        principal (float): Loan amount
        annual_rate (float): Annual interest rate (default 5%)
        years (int): Loan term in years (default 30)
    
    Returns:
        float: Monthly payment amount
    """
    monthly_rate = annual_rate / 12
    num_payments = years * 12
    
    if annual_rate == 0:
        return principal / num_payments
    
    payment = principal * (monthly_rate * (1 + monthly_rate)**num_payments) / \
              ((1 + monthly_rate)**num_payments - 1)
    
    return payment

# Different ways to call the function
loan_amount = 250000

# Use all defaults
payment1 = calculate_loan_payment(loan_amount)
print(f"Default terms: ${payment1:.2f}/month")

# Specify interest rate
payment2 = calculate_loan_payment(loan_amount, 0.065)
print(f"6.5% APR: ${payment2:.2f}/month")

# Specify all parameters
payment3 = calculate_loan_payment(loan_amount, 0.045, 15)
print(f"4.5% APR, 15 years: ${payment3:.2f}/month")

Single and Multiple Returns

def analyze_sales_performance(monthly_sales):
    """Analyze monthly sales data and return key metrics.
    
    Args:
        monthly_sales (list): List of monthly sales figures
    
    Returns:
        tuple: (total, average, best_month, worst_month)
    """
    total_sales = sum(monthly_sales)
    average_sales = total_sales / len(monthly_sales)
    best_month = max(monthly_sales)
    worst_month = min(monthly_sales)
    
    return total_sales, average_sales, best_month, worst_month

Variable Scope

Local vs Global Variables

# Global variables
company_name = "TechCorps Analytics"
tax_rate = 0.08

def calculate_total_cost(subtotal):
    """Calculate total cost including tax.
    
    Args:
        subtotal (float): Cost before tax
    
    Returns:
        float: Total cost including tax
    """
    # Local variable
    tax_amount = subtotal * tax_rate  # Uses global tax_rate
    total = subtotal + tax_amount      # Local variable
    
    # Can access global variables
    print(f"Processing order for {company_name}")
    return total

def process_order(items, prices):
    """Process an order and calculate totals."""
    # Local variables
    subtotal = 0
    
    print("Order Details:")
    for item, price in zip(items, prices):
        print(f"  {item}: ${price:.2f}")
        subtotal += price  # Local variable
    
    total = calculate_total_cost(subtotal)
    
    print(f"Subtotal: ${subtotal:.2f}")
    print(f"Tax ({tax_rate*100:.1f}%): ${subtotal * tax_rate:.2f}")
    print(f"Total: ${total:.2f}")
    
    return total

# Process a sample order
order_items = ["Laptop", "Mouse", "Keyboard"]
order_prices = [999.99, 29.99, 79.99]

final_total = process_order(order_items, order_prices)

def calculate_commission(sales):

sales_data = [12000, 15000, 8000, 22000, 18000, 9500]

high_performers = list(filter(lambda x: x >= 15000, sales_data))

Customer Loyalty Program System

def determine_loyalty_tier(total_spent, years_active):
    """Determine customer loyalty tier based on spending and tenure.
    
    Args:
        total_spent (float): Total amount spent by customer
        years_active (int): Years as active customer
    
    Returns:
        tuple: (tier, discount_rate, benefits)
    """
    if total_spent >= 10000 and years_active >= 3:
        return "Platinum", 0.15, ["Free shipping", "24/7 support", "Exclusive products"]
    elif total_spent >= 5000 and years_active >= 2:
        return "Gold", 0.10, ["Free shipping", "Priority support"]
    elif total_spent >= 1000 or years_active >= 1:
        return "Silver", 0.05, ["Free shipping on orders >$50"]
    else:
        return "Bronze", 0.00, ["Standard benefits"]

def process_customer_benefits(customers_data):
    """Process multiple customers and assign benefits."""
    results = []
    
    for customer_id, spent, years in customers_data:
        tier, discount, benefits = determine_loyalty_tier(spent, years)
        
        # Calculate potential savings
        annual_estimate = spent / max(years, 1)  # Avoid division by zero
        annual_savings = annual_estimate * discount
        
        results.append({
            'id': customer_id,
            'tier': tier,
            'discount': discount,
            'benefits': benefits,
            'annual_savings': annual_savings
        })
    
    return results

# Sample customer data: (customer_id, total_spent, years_active)
customers = [
    ("C001", 12500, 4),
    ("C002", 3200, 2),
    ("C003", 850, 1),
    ("C004", 15600, 5),
    ("C005", 2100, 1)
]

# Process customers
customer_analysis = process_customer_benefits(customers)

print("Customer Loyalty Analysis")
print("=" * 50)
for customer in customer_analysis:

Combining Concepts

Summit Exports Container System

def create_container_record(container_id, route, cargo_value):
    """Create immutable container record using tuple."""
    origin, destination = route  # Tuple unpacking
    return (container_id, origin, destination, cargo_value, "Active")

def track_destinations(containers):
    """Track unique destinations using set."""
    destinations = set()
    
    for container in containers:
        _, _, destination, _, _ = container  # Unpack tuple
        destinations.add(destination)
    
    return destinations

def calculate_route_statistics(containers):
    """Calculate statistics for container operations."""
    if not containers:
        return {}
    
    # Use sets to find unique values
    routes = set()
    total_value = 0
    
    for container in containers:
        container_id, origin, destination, value, status = container
        routes.add((origin, destination))
        total_value += value
    
    return {
        'total_containers': len(containers),
        'unique_routes': len(routes),
        'total_cargo_value': total_value,
        'average_value': total_value / len(containers),
        'routes': sorted(list(routes))  # Convert set back to sorted list
    }

# Create container records
container_data = [
    create_container_record("CONT001", ("Miami", "New York"), 15000),
    create_container_record("CONT002", ("Miami", "Boston"), 22000),
    create_container_record("CONT003", ("Tampa", "New York"), 18500),
    create_container_record("CONT004", ("Miami", "Boston"), 12000),
    create_container_record("CONT005", ("Jacksonville", "Boston"), 25000)
]

# Analyze operations
destinations = track_destinations(container_data)
stats = calculate_route_statistics(container_data)

print("Summit Exports Analysis")
print("=" * 30)
print(f"Active Destinations: {sorted(destinations)}")
print(f"Total Containers: {stats['total_containers']}")
print(f"Unique Routes: {stats['unique_routes']}")
print(f"Total Cargo Value: ${stats['total_cargo_value']:,.2f}")
print(f"Average Value: ${stats['average_value']:,.2f}")
print("\nActive Routes:")
for origin, destination in stats['routes']:
    print(f"  {origin} → {destination}")
