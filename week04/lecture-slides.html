def generate_business_report(title, **metrics):
    """Generate a formatted business report with flexible metrics.
    
    Args:
        title (str): Report title
        **metrics: Variable keyword arguments for business metrics
    """
    print(f"=" * 50)
    print(f"{title.upper()} REPORT")
    print(f"=" * 50)
    
    # Process each metric passed as keyword argument
    for metric_name, value in metrics.items():
        # Format metric name (replace underscores with spaces, capitalize)
        formatted_name = metric_name.replace('_', ' ').title()
        
        # Format value based on type
        if isinstance(value, (int, float)):
            if 'percentage' in metric_name.lower() or 'rate' in metric_name.lower():
                formatted_value = f"{value:.1f}%"
            elif 'revenue' in metric_name.lower() or 'cost' in metric_name.lower():
                formatted_value = f"${value:,.2f}"
            else:
                formatted_value = f"{value:,}"
        else:
            formatted_value = str(value)
        
        print(f"{formatted_name:.<30} {formatted_value}")
    
    print(f"=" * 50)

def comprehensive_sales_analysis(region, *monthly_sales, include_forecast=True, **additional_metrics):
    """Comprehensive sales analysis with all parameter types.
    
    Args:
        region (str): Geographic region for analysis
        *monthly_sales: Variable monthly sales figures
        include_forecast (bool): Whether to include forecast (default True)
        **additional_metrics: Additional business metrics
    
    Returns:
        dict: Analysis results
    """
    # Basic calculations
    total_sales = sum(monthly_sales)
    months_count = len(monthly_sales)
    average_monthly = total_sales / months_count if months_count > 0 else 0
    
    # Forecast if requested
    forecast = average_monthly * 1.05 if include_forecast else None
    
    print(f"\n{region.upper()} SALES ANALYSIS")
    print("-" * 40)
    print(f"Months Analyzed: {months_count}")
    print(f"Total Sales: ${total_sales:,.2f}")
    print(f"Average Monthly: ${average_monthly:,.2f}")
    
    if include_forecast:
        print(f"Next Month Forecast: ${forecast:,.2f}")
    
    # Display additional metrics
    if additional_metrics:
        print("\nAdditional Metrics:")
        for metric, value in additional_metrics.items():
            metric_name = metric.replace('_', ' ').title()
            if isinstance(value, (int, float)):
                print(f"  {metric_name}: {value:,.2f}")
            else:
                print(f"  {metric_name}: {value}")
    
    return {
        'region': region,
        'total_sales': total_sales,
        'average_monthly': average_monthly,
        'forecast': forecast,
        'additional_metrics': additional_metrics
    }

def count_business_days(start_date, end_date):
    """Count business days between two dates (excludes weekends)."""
    business_days = 0
    current_date = start_date
    
    while current_date <= end_date:
        if current_date.weekday() < 5:  # Monday is 0, Friday is 4
            business_days += 1
        current_date += datetime.timedelta(days=1)

"""
Business Utilities Module
Provides common business calculation functions
"""

def calculate_roi(initial_investment, final_value):
    """Calculate Return on Investment (ROI).
    
    Args:
        initial_investment (float): Initial investment amount
        final_value (float): Final value of investment
    
    Returns:
        float: ROI as a percentage
    """
    if initial_investment == 0:
        return 0
    
    roi = ((final_value - initial_investment) / initial_investment) * 100
    return roi

def calculate_break_even_point(fixed_costs, variable_cost_per_unit, price_per_unit):
    """Calculate break-even point in units.
    
    Args:
        fixed_costs (float): Total fixed costs
        variable_cost_per_unit (float): Variable cost per unit
        price_per_unit (float): Selling price per unit
    
    Returns:
        float: Break-even point in units
    """
    if price_per_unit <= variable_cost_per_unit:
        raise ValueError("Price must be greater than variable cost per unit")
    
    contribution_margin = price_per_unit - variable_cost_per_unit
    break_even_units = fixed_costs / contribution_margin
    return break_even_units

def format_currency(amount, currency_symbol="$"):
    """Format amount as currency string.
    
    Args:
        amount (float): Amount to format
        currency_symbol (str): Currency symbol to use
    
    Returns:
        str: Formatted currency string
    """
    return f"{currency_symbol}{amount:,.2f}"

def calculate_compound_interest(principal, rate, time, compounding_frequency=12):
    """Calculate compound interest.
    
    Args:
        principal (float): Initial principal amount
        rate (float): Annual interest rate (as decimal)
        time (float): Time in years
        compounding_frequency (int): Number of times interest is compounded per year
    
    Returns:
        float: Final amount after compound interest
    """
    amount = principal * (1 + rate/compounding_frequency)**(compounding_frequency * time)
    return amount

def calculate_product_metrics(sales_data, costs_data):
    """Calculate comprehensive product performance metrics."""
    metrics = {}
    
    for product, sales in sales_data.items():
        cost = costs_data.get(product, 0)
        
        # Basic calculations
        profit = sales - cost
        margin = (profit / sales * 100) if sales > 0 else 0
        roi = (profit / cost * 100) if cost > 0 else 0
        
        metrics[product] = {
            'sales': sales,
            'cost': cost,
            'profit': profit,
            'margin': margin,
            'roi': roi
        }
    
    return metrics

Making Your Functions More Robust

def safe_calculate_metrics(revenue, costs):
    """Calculate business metrics with error handling."""
    try:
        # Input validation
        if not isinstance(revenue, (int, float)) or not isinstance(costs, (int, float)):
            raise TypeError("Revenue and costs must be numeric values")
        
        if revenue < 0 or costs < 0:
            raise ValueError("Revenue and costs cannot be negative")
        
        if costs == 0:
            print("Warning: Costs are zero, ROI calculation may not be meaningful")
        
        # Calculations
        profit = revenue - costs
        margin = (profit / revenue) * 100 if revenue > 0 else 0
        roi = (profit / costs) * 100 if costs > 0 else float('inf')
        
        return {
            'profit': profit,
            'margin': margin,
            'roi': roi,
            'status': 'success'
        }
    
    except TypeError as e:
        return {'status': 'error', 'message': f"Type error: {e}"}
    except ValueError as e:
        return {'status': 'error', 'message': f"Value error: {e}"}
    except Exception as e:
        return {'status': 'error', 'message': f"Unexpected error: {e}"}
