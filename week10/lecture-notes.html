Example: Two-Sample t-test

# Comparing sales between two stores
store_a = np.random.normal(100, 15, 30)
store_b = np.random.normal(110, 15, 30)

# Perform t-test
t_stat, p_value = stats.ttest_ind(store_a, store_b)
print(f"t-statistic: {t_stat:.3f}")
print(f"p-value: {p_value:.3f}")

# Interpret results
alpha = 0.05
if p_value < alpha:
    print("Reject null hypothesis: Significant difference exists")
else:
    print("Fail to reject null hypothesis: No significant difference")
