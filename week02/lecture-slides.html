The if Statement

Basic Syntax

Adding else

Multiple Conditions with elif

# Customer categorization based on spending
customer_spending = 7500

if customer_spending >= 10000:
    category = "Platinum"
    discount = 0.20
elif customer_spending >= 5000:
    category = "Gold"
    discount = 0.15
elif customer_spending >= 1000:
    category = "Silver"
    discount = 0.10
else:
    category = "Bronze"
    discount = 0.05

print(f"Customer Category: {category}")
print(f"Discount: {discount*100}%")
Customer Category: Gold
Discount: 15.0%

Combining Conditions

# Using and, or, not
age = 25
income = 55000
credit_score = 720

# Loan approval logic
if age >= 18 and income >= 50000 and credit_score >= 700:
    print("Loan approved!")

Remember: and requires ALL conditions true, or requires ANY condition true

The while Loop

Repeating Until a Condition

# Compound interest calculation
principal = 1000
rate = 0.05
target = 1500
years = 0

while principal < target:
    principal *= (1 + rate)
    years += 1
    print(f"Year {years}: ${principal:.2f}")

print(f"It takes {years} years to reach ${target}")

Warning

Always ensure your while loop condition will eventually become False!

Loop Control: break and continue

# Finding the first problem transaction
transactions = [100, 250, -50, 300, -25, 400]

for i, amount in enumerate(transactions):
    if amount < 0:
        print(f"Problem found at transaction {i+1}: ${amount}")
        break  # Exit the loop
    print(f"Transaction {i+1}: OK")
Transaction 1: OK
Transaction 2: OK
Problem found at transaction 3: $-50
# Skip negative values
for amount in transactions:
    if amount < 0:
        continue  # Skip to next iteration
    print(f"Processing payment: ${amount}")
Processing payment: $100
Processing payment: $250
Processing payment: $300
Processing payment: $400

Part 2: Lists

Introduction to Lists

Lists are: - Ordered collections of items - Mutable (can be changed) - Versatile (can hold any data type)

empty_list = []
numbers = [1, 2, 3, 4, 5]

List Indexing and Slicing

products = ["Laptop", "Mouse", "Keyboard", "Monitor", "Webcam"]

# Indexing (0-based)
print(f"First item: {products[0]}")
print(f"Last item: {products[-1]}")

# Slicing [start:end:step]
print(f"First three: {products[:3]}")
print(f"Last two: {products[-2:]}")
print(f"Every other: {products[::2]}")
First item: Laptop
Last item: Webcam
First three: ['Laptop', 'Mouse', 'Keyboard']
Last two: ['Monitor', 'Webcam']
Every other: ['Laptop', 'Keyboard', 'Webcam']

Note

Remember: Slicing creates a new list, it doesn’t modify the original

List Methods

Adding Elements

inventory = ["Laptop", "Mouse"]

Removing Elements

List Comprehensions

Elegant List Creation

# Traditional way
squares = []
for i in range(1, 6):
    squares.append(i ** 2)
print(f"Traditional: {squares}")

# List comprehension
squares_comp = [i ** 2 for i in range(1, 6)]
print(f"Comprehension: {squares_comp}")

# With condition
evens = [x for x in range(10) if x % 2 == 0]
print(f"Even numbers: {evens}")

# Business example: Calculate discounted prices
prices = [100, 250, 75, 300]
discounted = [p * 0.9 for p in prices if p >= 100]
print(f"Discounted prices: {discounted}")
Traditional: [1, 4, 9, 16, 25]
Comprehension: [1, 4, 9, 16, 25]
Even numbers: [0, 2, 4, 6, 8]
Discounted prices: [90.0, 225.0, 270.0]

Introduction to Dictionaries

Dictionaries store key-value pairs: - Keys must be unique and immutable - Values can be any type - Unordered (Python 3.7+ maintains insertion order)

# Creating dictionaries

Accessing Dictionary Values

employee = {
    "name": "John Smith",
    "department": "Sales",
    "salary": 65000,
    "years": 3

# Direct access
print(f"Name: {employee['name']}")

print(f"Department: {employee.get('department')}")

Combining Concepts

Real Business Example

# E-commerce order processing
orders = [
    {"id": 1001, "customer": "Alice", "total": 250, "status": "pending"},
    {"id": 1002, "customer": "Bob", "total": 75, "status": "shipped"},
    {"id": 1003, "customer": "Charlie", "total": 500, "status": "pending"},
    {"id": 1004, "customer": "Alice", "total": 125, "status": "delivered"}
]

# Process orders
pending_total = 0
customer_totals = {}

for order in orders:
    # Calculate pending revenue
    if order["status"] == "pending":
        pending_total += order["total"]
    
    # Track spending by customer
    customer = order["customer"]
    if customer in customer_totals:
        customer_totals[customer] += order["total"]
    else:
        customer_totals[customer] = order["total"]

print(f"Pending revenue: ${pending_total}")
print(f"Customer totals: {customer_totals}")
Pending revenue: $750
Customer totals: {'Alice': 375, 'Bob': 75, 'Charlie': 500}

List and Dictionary Comprehensions

Advanced Data Processing

# Sales data
sales_data = [
    {"product": "Laptop", "quantity": 5, "price": 999},
    {"product": "Mouse", "quantity": 20, "price": 25},
    {"product": "Keyboard", "quantity": 10, "price": 75},
    {"product": "Monitor", "quantity": 3, "price": 299}
]

# Calculate revenues with list comprehension
revenues = [item["quantity"] * item["price"] for item in sales_data]
print(f"Revenues: {revenues}")

# Create price lookup with dictionary comprehension
price_lookup = {item["product"]: item["price"] for item in sales_data}
print(f"Price lookup: {price_lookup}")

# Filter high-value sales
high_value = [item for item in sales_data 
              if item["quantity"] * item["price"] > 500]
print(f"High value sales: {[item['product'] for item in high_value]}")
Revenues: [4995, 500, 750, 897]
Price lookup: {'Laptop': 999, 'Mouse': 25, 'Keyboard': 75, 'Monitor': 299}
High value sales: ['Laptop', 'Keyboard', 'Monitor']
