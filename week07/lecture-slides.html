# Sales performance dataset
sales_data = {
    'Region': ['North', 'South', 'East', 'West', 'North', 'South', 'East', 'West'],

Handling Missing Data

# Create customer data with missing values
customer_data = {
    'Customer_ID': ['C001', 'C002', 'C003', 'C004', 'C005', 'C006'],
    'Name': ['John Doe', 'Jane Smith', None, 'Bob Johnson', 'Alice Wilson', 'Carol Brown'],
    'Age': [35, 28, 42, None, 31, 39],
    'Income': [75000, 65000, 85000, 70000, None, 90000],
    'Purchase_Amount': [1200, None, 2100, 850, 950, 1800],
    'City': ['Tampa', 'Miami', 'Orlando', 'Tampa', None, 'Jacksonville']
}

df_customers = pd.DataFrame(customer_data)

print("CUSTOMER DATA CLEANING")
print("=" * 25)
print("Original Data:")
print(df_customers)

# Identify missing data
print(f"\nMissing Data Analysis:")
print(df_customers.isnull().sum())
print(f"\nMissing Data Percentage:")
print((df_customers.isnull().sum() / len(df_customers) * 100).round(2))

# Different strategies for handling missing data

# 1. Drop rows with any missing values
print(f"\nStrategy 1 - Drop All Missing (Complete Cases Only):")
clean_all = df_customers.dropna()
print(f"Remaining rows: {len(clean_all)} (lost {len(df_customers) - len(clean_all)} rows)")
print(clean_all)

# 2. Drop rows with missing critical information (Name)
print(f"\nStrategy 2 - Drop Missing Names Only:")
clean_names = df_customers.dropna(subset=['Name'])
print(f"Remaining rows: {len(clean_names)} (lost {len(df_customers) - len(clean_names)} rows)")

# 3. Fill missing values with appropriate replacements
print(f"\nStrategy 3 - Fill Missing Values:")
df_filled = df_customers.copy()

# Fill numeric missing values
df_filled['Age'].fillna(df_filled['Age'].mean(), inplace=True)
df_filled['Income'].fillna(df_filled['Income'].median(), inplace=True)
df_filled['Purchase_Amount'].fillna(df_filled['Purchase_Amount'].mean(), inplace=True)

# Fill categorical missing values
df_filled['Name'].fillna('Unknown Customer', inplace=True)
df_filled['City'].fillna('Not Specified', inplace=True)

print(df_filled)

# Business impact analysis
print(f"\nBusiness Impact Analysis:")
original_revenue = df_customers['Purchase_Amount'].sum()
cleaned_revenue = df_filled['Purchase_Amount'].sum()
print(f"Original Total Revenue: ${original_revenue:,.2f}")
print(f"After Cleaning: ${cleaned_revenue:,.2f}")
print(f"Data Recovery: {(cleaned_revenue/original_revenue)*100:.1f}%")

product_sales['Product_Clean'] = product_sales['Product'].str.title()

Part 4: Grouping and Aggregation

GroupBy Operations

# Comprehensive sales dataset
sales_detail = pd.DataFrame({
    'Date': pd.date_range('2024-01-01', periods=200, freq='D'),
    'Region': np.random.choice(['North', 'South', 'East', 'West'], 200),
    'Product': np.random.choice(['Laptop', 'Tablet', 'Phone', 'Accessories'], 200),
    'Sales_Rep': np.random.choice(['Alice', 'Bob', 'Carol', 'David', 'Eve'], 200),
    'Revenue': np.random.normal(5000, 1500, 200),
    'Units_Sold': np.random.poisson(10, 200)
})

# Ensure positive values
sales_detail['Revenue'] = np.abs(sales_detail['Revenue'])
sales_detail['Units_Sold'] = np.abs(sales_detail['Units_Sold']) + 1

# Add derived columns
sales_detail['Month'] = sales_detail['Date'].dt.strftime('%B')
sales_detail['Quarter'] = sales_detail['Date'].dt.quarter

print("GROUPBY OPERATIONS ANALYSIS")
print("=" * 35)
print(f"Sample Data ({len(sales_detail)} records):")
print(sales_detail.head())

# Single column grouping
print(f"\nRevenue by Region:")
revenue_by_region = sales_detail.groupby('Region')['Revenue'].agg(['sum', 'mean', 'count'])
revenue_by_region.columns = ['Total_Revenue', 'Avg_Revenue', 'Transactions']
revenue_by_region['Total_Revenue'] = revenue_by_region['Total_Revenue'].round(2)
revenue_by_region['Avg_Revenue'] = revenue_by_region['Avg_Revenue'].round(2)
print(revenue_by_region)

# Multiple column grouping
print(f"\nRevenue by Region and Product:")
region_product = sales_detail.groupby(['Region', 'Product'])['Revenue'].sum().reset_index()
# Show top 10 combinations
top_combinations = region_product.nlargest(10, 'Revenue')
print(top_combinations)

# Multiple aggregations
print(f"\nSales Rep Performance Analysis:")
rep_performance = sales_detail.groupby('Sales_Rep').agg({
    'Revenue': ['sum', 'mean', 'std'],
    'Units_Sold': ['sum', 'mean'],
    'Date': 'count'  # Count of transactions
}).round(2)

# Flatten column names
rep_performance.columns = ['Total_Revenue', 'Avg_Revenue', 'Revenue_Std', 
                          'Total_Units', 'Avg_Units', 'Transactions']
print(rep_performance)

# Identify top performers
print(f"\nTop 3 Sales Reps by Total Revenue:")
top_reps = rep_performance.nlargest(3, 'Total_Revenue')
for rep, data in top_reps.iterrows():
    consistency = data['Revenue_Std'] / data['Avg_Revenue']  # Coefficient of variation
    print(f"  {rep}: ${data['Total_Revenue']:,.2f} total, "
          f"${data['Avg_Revenue']:,.2f} avg, "
          f"{consistency:.2f} consistency ratio")

    'Customer_ID': np.random.choice(customers['Customer_ID'], n_transactions),
    'Transaction_Date': pd.date_range('2023-01-01', '2023-12-31', periods=n_transactions),
    'Product_Category': np.random.choice(['Software', 'Consulting', 'Training', 'Support'], n_transactions),
    'Amount': np.random.lognormal(8, 1, n_transactions),  # Log-normal for realistic revenue distribution
    'Quantity': np.random.poisson(3, n_transactions) + 1

print(f"  Transactions: {len(transactions):,}")

}).round(2)

# Calculate additional metrics
customer_metrics['Days_Active'] = (customer_metrics['Last_Purchase'] - customer_metrics['First_Purchase']).dt.days
customer_metrics['Revenue_per_Day'] = (customer_metrics['Total_Revenue'] / 
                                      customer_metrics['Days_Active'].replace(0, 1)).round(2)

# Merge back with customer info
customer_analysis = pd.merge(customer_metrics.reset_index(), customers, on='Customer_ID')
